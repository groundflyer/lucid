104  noexcept
105  {
106  const auto& [sample_pos, sample_val] = sample;
107  const real weight = filter(pixel_ssp, sample_pos);
108  RGB old_color(old_rgba);
109  const real old_weight = old_rgba.template get<3>();
110  const real new_weight = old_weight + weight;
111  if(new_weight > 0_r)
112  old_color = (old_color * old_weight + sample_val * weight) / new_weight;
113  return RGBA(old_color, new_weight);
114  }
