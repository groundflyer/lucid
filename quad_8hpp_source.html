22 constexpr Intersection
23 intersect(const Ray_<RayContainer>& ray, const Quad_<QuadContainer>& prim) noexcept
24 {
25  const auto& [o, d] = ray;
26  const auto& [v00, v01, v11, v10] = prim;
27  const auto e01 = v10 - v00;
28  const auto e03 = v01 - v00;
29  const auto p = d.cross(e03);
30  const real D = e01.dot(p);
31 
32  if(math::abs(D) < std::numeric_limits<real>::min()) return Intersection();
33 
34  const auto T = o - v00;
35  const auto alpha = T.dot(p) / D;
36 
37  const constexpr auto range01 = range(0_r, 1_r);
38 
39  if(!range01(alpha)) return Intersection();
40 
41  const auto Q = T.cross(e01);
42  const auto beta = d.dot(Q) / D;
43  if(!range01(beta)) return Intersection();
44 
45  if((alpha + beta) > 1)
46  {
47  const auto e23 = v01 - v11;
48  const auto e21 = v10 - v11;
49  const auto p_ = d.cross(e21);
50  const auto D_ = e23.dot(p_);
51 
52  if(math::abs(D_) < std::numeric_limits<real>::min()) return Intersection();
53 
54  const auto T_ = o - v11;
55  const auto alpha_ = T_.dot(p_) / D_;
56 
57  if(alpha_ < 0) return Intersection();
58 
59  const auto Q_ = T_.cross(e23);
60  const auto beta_ = d.dot(Q_) / D_;
61  if(beta_ < 0) return Intersection();
62  }
63 
64  const auto t = e03.dot(Q) / D;
65 
66  if(t < 0) return Intersection();
67 
68  const auto e02 = v11 - v00;
69  const auto N = e01.cross(e03);
70  const auto aN = abs(N);
71  const auto& [Nx, Ny, Nz] = N;
72  const auto& [aNx, aNy, aNz] = aN;
73 
74  real a11{}, b11{};
75  if((aNx >= aNy) && (aNx >= aNz))
76  {
77  a11 = (e02[1] * e03[2] - e02[2] * e03[1]) / Nx;
78  b11 = (e01[1] * e02[2] - e01[2] * e02[1]) / Nx;
79  }
80  else if((aNy >= aNx) && (aNy >= aNz))
81  {
82  a11 = (e02[2] * e03[0] - e02[0] * e03[2]) / Ny;
83  b11 = (e01[2] * e02[0] - e01[0] * e02[2]) / Ny;
84  }
85  else
86  {
87  a11 = (e02[0] * e03[1] - e02[1] * e03[0]) / Nz;
88  b11 = (e01[0] * e02[1] - e01[1] * e02[0]) / Nz;
89  }
90 
91  real u{}, v{};
92  if(math::abs(a11 - 1) < std::numeric_limits<real>::min())
93  {
94  u = alpha;
95  if(math::abs(b11 - 1) < std::numeric_limits<real>::min())
96  v = beta;
97  else
98  v = beta / (u * (b11 - 1) + 1);
99  }
100  else if(math::abs(b11 - 1) < std::numeric_limits<real>::min())
101  {
102  v = beta;
103  u = alpha / (v * (a11 - 1) + 1);
104  }
105  else
106  {
107  const auto A = -(b11 - 1);
108  const auto B = alpha * (b11 - 1) - beta * (a11 - 1) - 1;
109  const auto C = alpha;
110  const auto Delta = B * B - 4 * A * C;
111  const auto QQ = -0.5_r * (B + std::copysign(math::sqrt(Delta), B));
112  u = QQ / A;
113 
114  if(!range01(u)) u = C / QQ;
115 
116  v = beta / (u * (b11 - 1) + 1);
117  }
118 
119  return Intersection{true, t, Vec2(u, v)};
120 }
