50 template <template <typename, size_t> typename AABBContainer,
51  template <typename, size_t>
52  typename RayContainer>
53 constexpr Intersection
54 intersect(const Ray_<RayContainer>& ray, const AABB_<AABBContainer>& prim) noexcept
55 {
56  const auto& [o, d] = ray;
57  const auto inv_d = Vec3{1_r} / d;
58  const auto sign = inv_d < 0_r;
59  const auto vmin = inv_d * (prim[sign] - o);
60  const auto vmax = inv_d * (prim[!sign] - o);
61  const auto& [xmin, ymin, zmin] = vmin;
62  const auto& [xmax, ymax, zmax] = vmax;
63  const real& tmin1 = std::max(ymin, xmin);
64  const real& tmax1 = std::min(ymax, xmax);
65  const real& tmin2 = std::max(zmin, tmin1);
66  const bool intersected = xmin <= ymax && ymin <= xmax && tmin1 <= zmax && zmin <= tmax1;
67  return Intersection{intersected && !std::signbit(tmin2), tmin2, Vec2{}};
68 }

71  template <typename, size_t>
72  typename RayContainer,
73  template <typename, size_t>
74  typename IsectContainer>
75 constexpr Normal
76 normal(const Ray_<RayContainer>& ray,
77  const Intersection_<IsectContainer>& isect,
78  const AABB_<AABBContainer>& prim) noexcept
79 {
80  const auto& [o, d] = ray;
81  const Point pos = o + d * isect.t;
82  const real nsign[2]{-1_r, 1_r};
83  Vec3 vd[2]{};
84  real md[2]{};
85  for(unsigned i = 0; i < 2; ++i)
86  {
87  const auto ad = lucid::abs(pos - prim[bool(i)]);
88  md[i] = lucid::min(ad);
89  vd[i] = Vec3(ad == md[i]);
90  }
91  const auto pp = md[0] > md[1];
92  return Normal(vd[pp] * nsign[pp]);
93 }
